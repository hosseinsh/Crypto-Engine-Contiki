225 int
226 dtls_get_cookie(uint8 *msg, int msglen, uint8 **cookie) {
227  /* To access the cookie, we have to determine the session id's
228  * length and skip the whole thing. */
229  if (msglen < DTLS_HS_LENGTH + DTLS_CH_LENGTH + sizeof(uint8)
230  || dtls_uint16_to_int(msg + DTLS_HS_LENGTH) != DTLS_VERSION)
231  return -1;
232  msglen -= DTLS_HS_LENGTH + DTLS_CH_LENGTH;
233  msg += DTLS_HS_LENGTH + DTLS_CH_LENGTH;
234 
235  SKIP_VAR_FIELD(msg, msglen, uint8); /* skip session id */
236 
237  if (msglen < (*msg & 0xff) + sizeof(uint8))
238  return -1;
239 
240  *cookie = msg + sizeof(uint8);
241  return dtls_uint8_to_int(msg);
242 
243  error:
244  return -1;
245 }

247 int
248 dtls_create_cookie(dtls_context_t *ctx,
249  session_t *session,
250  uint8 *msg, int msglen,
251  uint8 *cookie, int *clen) {
252  unsigned char buf[DTLS_HMAC_MAX];
253  size_t len, e;
254 
255  /* create cookie with HMAC-SHA256 over:
256  * - SECRET
257  * - session parameters (only IP address?)
258  * - client version
259  * - random gmt and bytes
260  * - session id
261  * - cipher_suites
262  * - compression method
263  */
264 
265  /* We use our own buffer as hmac_context instead of a dynamic buffer
266  * created by dtls_hmac_new() to separate storage space for cookie
267  * creation from storage that is used in real sessions. Note that
268  * the buffer size must fit with the default hash algorithm (see
269  * implementation of dtls_hmac_context_new()). */
270 
271  dtls_hmac_context_t hmac_context;
272  dtls_hmac_init(&hmac_context, ctx->cookie_secret, DTLS_COOKIE_SECRET_LENGTH);
273 
274  dtls_hmac_update(&hmac_context,
275  (unsigned char *)&session->addr, session->size);
276 
277  /* feed in the beginning of the Client Hello up to and including the
278  session id */
279  e = sizeof(dtls_client_hello_t);
280  e += (*(msg + DTLS_HS_LENGTH + e) & 0xff) + sizeof(uint8);
281 
282  dtls_hmac_update(&hmac_context, msg + DTLS_HS_LENGTH, e);
283 
284  /* skip cookie bytes and length byte */
285  e += *(uint8 *)(msg + DTLS_HS_LENGTH + e) & 0xff;
286  e += sizeof(uint8);
287 
288  dtls_hmac_update(&hmac_context,
289  msg + DTLS_HS_LENGTH + e,
290  dtls_get_fragment_length(DTLS_HANDSHAKE_HEADER(msg)) - e);
291 
292  len = dtls_hmac_finalize(&hmac_context, buf);
293 
294  if (len < *clen) {
295  memset(cookie + len, 0, *clen - len);
296  *clen = len;
297  }
298 
299  memcpy(cookie, buf, *clen);
300  return 1;
301 }

423 int
424 dtls_verify_peer(dtls_context_t *ctx,
425  dtls_peer_t *peer,
426  session_t *session,
427  uint8 *record,
428  uint8 *data, size_t data_length) {
429 
430  int len = DTLS_COOKIE_LENGTH;
431  uint8 *cookie, *p;
432 #undef mycookie
433 #define mycookie (ctx->sendbuf + HV_HDR_LENGTH)
434 
435  /* check if we can access at least all fields from the handshake header */
436  if (record[0] == DTLS_CT_HANDSHAKE
437  && data_length >= DTLS_HS_LENGTH
438  && data[0] == DTLS_HT_CLIENT_HELLO) {
439 
440  /* Store cookie where we can reuse it for the HelloVerify request. */
441  if (dtls_create_cookie(ctx, session, data, data_length,
442  mycookie, &len) < 0)
443  return -1;
444 /* #ifndef NDEBUG */
445 /* debug("create cookie: "); */
446 /* dump(mycookie, len); */
447 /* printf("\n"); */
448 /* #endif */
449  assert(len == DTLS_COOKIE_LENGTH);
450 
451  /* Perform cookie check. */
452  len = dtls_get_cookie(data, data_length, &cookie);
453 
454 /* #ifndef NDEBUG */
455 /* debug("compare with cookie: "); */
456 /* dump(cookie, len); */
457 /* printf("\n"); */
458 /* #endif */
459 
460  /* check if cookies match */
461  if (len == DTLS_COOKIE_LENGTH && memcmp(cookie, mycookie, len) == 0) {
462  debug("found matching cookie\n");
463  return 1;
464  }
465  if (len > 0) {
466  debug("invalid cookie");
467 #ifndef NDEBUG
468  dump(cookie, len);
469  printf("\n");
470 #endif
471  }
472  /* ClientHello did not contain any valid cookie, hence we send a
473  * HelloVerify request. */
474 
475  p = dtls_set_handshake_header(DTLS_HT_HELLO_VERIFY_REQUEST,
476  peer, DTLS_HV_LENGTH + DTLS_COOKIE_LENGTH,
477  0, DTLS_HV_LENGTH + DTLS_COOKIE_LENGTH,
478  ctx->sendbuf + DTLS_RH_LENGTH);
479 
480  dtls_int_to_uint16(p, DTLS_VERSION);
481  p += sizeof(uint16);
482 
483  dtls_int_to_uint8(p, DTLS_COOKIE_LENGTH);
484  p += sizeof(uint8);
485 
486  assert(p == mycookie);
487 
488  p += DTLS_COOKIE_LENGTH;
489 
490  if (!peer) {
491  /* It's an initial ClientHello, so we set the record header
492  * manually and send the HelloVerify request using the
493  * registered write callback. */
494 
495  dtls_set_record_header(DTLS_CT_HANDSHAKE, NULL, ctx->sendbuf);
496  /* set packet length */
497  dtls_int_to_uint16(ctx->sendbuf + 11,
498  p - (ctx->sendbuf + DTLS_RH_LENGTH));
499 
500  (void)CALL(ctx, write, session, ctx->sendbuf, p - ctx->sendbuf);
501  } else {
502  if (peer->epoch) {
503  debug("renegotiation, therefore we accept it anyway:");
504  return 1;
505  }
506 
507  if (dtls_send(ctx, peer, DTLS_CT_HANDSHAKE,
508  ctx->sendbuf + DTLS_RH_LENGTH,
509  p - (ctx->sendbuf + DTLS_RH_LENGTH)) < 0) {
510  warn("cannot send HelloVerify request\n");
511  return -1;
512  }
513  }
514 
515  return 0; /* HelloVerify is sent, now we cannot do anything but wait */
516  }
517 
518  return -1; /* not a ClientHello, signal error */
519 #undef mycookie
520 }

705  config->cipher = CURRENT_CONFIG(peer)->cipher;
706  config->compression = CURRENT_CONFIG(peer)->compression;

709  }
710

712  data_length -= sizeof(uint16) + i;
713 
714  ok = 0;

728  return 0;
729  }
730

734  return 1;
735  }
736

738  if (data_length < i + sizeof(uint8))
739  goto error;
740 
741  data += sizeof(uint8);
742  data_length -= sizeof(uint8) + i;

753  }

930 dtls_prepare_record(dtls_peer_t *peer,
931  unsigned char type,
932  uint8 *data, size_t data_length,

938  if (*rlen < DTLS_RH_LENGTH + data_length) {
939  debug("dtls_prepare_record: send buffer too small\n");
940  return -1;
941  }
942 
943  p = dtls_set_record_header(type, peer, sendbuf);
944 
945  if (CURRENT_CONFIG(peer)->cipher == TLS_NULL_WITH_NULL_NULL) {
946  /* no cipher suite */
947  memcpy(p, data, data_length);
948  res = data_length;
949  } else { /* TLS_PSK_WITH_AES_128_CCM_8 */
950  dtls_cipher_context_t *cipher_context;
951 
956 #define A_DATA_LEN 13
957 #define A_DATA N
958  unsigned char N[max(DTLS_CCM_BLOCKSIZE, A_DATA_LEN)];
959 
960  if (*rlen < sizeof(dtls_record_header_t) + data_length + 8) {
961  warn("dtls_prepare_record(): send buffer too small\n");
962  return -1;
963  }
964 
965  debug("dtls_prepare_record(): encrypt using TLS_PSK_WITH_AES_128_CCM_8\n");
966 
967  /* set nonce
968  from http://tools.ietf.org/html/draft-mcgrew-tls-aes-ccm-03:
969  struct {
970  case client:
971  uint32 client_write_IV; // low order 32-bits
972  case server:
973  uint32 server_write_IV; // low order 32-bits
974  uint64 seq_num;
975  } CCMNonce.
976 
977  In DTLS, the 64-bit seq_num is the 16-bit epoch concatenated with the
978  48-bit seq_num.
979  */
980 
981  memcpy(p, &DTLS_RECORD_HEADER(sendbuf)->epoch, 8);
982  memcpy(p + 8, data, data_length);
983 
984  memset(N, 0, DTLS_CCM_BLOCKSIZE);
985  memcpy(N, dtls_kb_local_iv(CURRENT_CONFIG(peer)),
986  dtls_kb_iv_size(CURRENT_CONFIG(peer)));
987  memcpy(N + dtls_kb_iv_size(CURRENT_CONFIG(peer)), p, 8); /* epoch + seq_num */
988 
989  cipher_context = CURRENT_CONFIG(peer)->write_cipher;
990 
991  if (!cipher_context) {
992  warn("no write_cipher available!\n");
993  return -1;
994  }
995 /* #ifndef NDEBUG */
996 /* printf("nonce:\t"); */
997 /* dump(N, DTLS_CCM_BLOCKSIZE); */
998 /* printf("\nkey:\t"); */
999 /* dump(dtls_kb_local_write_key(CURRENT_CONFIG(peer)), */
1000 /* dtls_kb_key_size(CURRENT_CONFIG(peer))); */
1001 /* printf("\n"); */
1002 /* #endif */

1004 
1005  /* re-use N to create additional data according to RFC 5246, Section 6.2.3.3:
1006  *
1007  * additional_data = seq_num + TLSCompressed.type +
1008  * TLSCompressed.version + TLSCompressed.length;
1009  */
1010  memcpy(A_DATA, &DTLS_RECORD_HEADER(sendbuf)->epoch, 8); /* epoch and seq_num */
1011  memcpy(A_DATA + 8, &DTLS_RECORD_HEADER(sendbuf)->content_type, 3); /* type and version */
1012  dtls_int_to_uint16(A_DATA + 11, data_length); /* length */
1013 
1014  res = dtls_encrypt(cipher_context, p + 8, data_length, p + 8,
1015  A_DATA, A_DATA_LEN);
1016 
1017  if (res < 0)
1018  return -1;

1021 /* dump(p, res + 8); */
1022 /* printf("\n"); */
1023 /* #endif */

1025  }
1026

1028  dtls_int_to_uint16(sendbuf + 11, res);
1029 
1030  *rlen = DTLS_RH_LENGTH + res;
1031  return 1;
1032 }

1635

1656  dtls_kb_iv_size(CURRENT_CONFIG(peer)));

1671 /* printf("nonce:\t"); */

1705 /* printf("\ncleartext:\n"); */
1706 /* dump(*cleartext, *clen); */
1707 /* printf("\n"); */
1708 /* #endif */
1709  }
1710 
1711  return ok;
1712 }

1717  uint8 *record_header, uint8 *data, size_t data_length) {
1718

1720  * respect to the current internal state for this peer. In case of
1721  * error, it is left with return 0. */

1724 
1725  /************************************************************************
1726  * Client states
1727  ************************************************************************/
1728 
1729  case DTLS_STATE_CLIENTHELLO:
1730  /* here we expect a HelloVerify or ServerHello */

1733  if (check_server_hello(ctx, peer, data, data_length)) {
1734  peer->state = DTLS_STATE_WAIT_SERVERHELLODONE;
1735  /* update_hs_hash(peer, data, data_length); */
1736  }
1737 
1738  break;
1739 
1740  case DTLS_STATE_WAIT_SERVERHELLODONE:
1741  /* expect a ServerHelloDone */
1742 
1743  debug("DTLS_STATE_WAIT_SERVERHELLODONE\n");
1744 
1745  if (check_server_hellodone(ctx, peer, data, data_length)) {
1746  peer->state = DTLS_STATE_WAIT_SERVERFINISHED;
1747  /* update_hs_hash(peer, data, data_length); */
1748  }
1749 
1750  break;
1751 
1752  case DTLS_STATE_WAIT_SERVERFINISHED:
1753  /* expect a Finished message from server */
1754 
1755  debug("DTLS_STATE_WAIT_SERVERFINISHED\n");
1756  if (check_finished(ctx, peer, record_header, data, data_length)) {
1757  debug("finished!\n");
1758  peer->state = DTLS_STATE_CONNECTED;
1759  }
1760 
1761  break;
1762 
1763  /************************************************************************
1764  * Server states
1765  ************************************************************************/
1766 
1767  case DTLS_STATE_SERVERHELLO:

1769  /* handle ClientHello, update msg and msglen and goto next if not finished */
1770 
1771  debug("DTLS_STATE_SERVERHELLO\n");
1772  if (!check_client_keyexchange(ctx, peer, data, data_length)) {
1773  warn("check_client_keyexchange failed (%d, %d)\n", data_length, data[0]);

1778  peer->state = DTLS_STATE_KEYEXCHANGE;
1779  break;
1780 
1781  case DTLS_STATE_WAIT_FINISHED:

1784  debug("finished!\n");
1785 
1786  /* send ServerFinished */
1787  update_hs_hash(peer, data, data_length);
1788 
1789  if (dtls_send_server_finished(ctx, peer) > 0) {
1790  peer->state = DTLS_STATE_CONNECTED;
1791  } else {
1792  warn("sending server Finished failed\n");
1793  }
1794  } else {
1795  /* send alert */
1796  }
1797  break;
1798 
1799  case DTLS_STATE_CONNECTED:
1800  /* At this point, we have a good relationship with this peer. This
1801  * state is left for re-negotiation of key material. */
1802 
1803  debug("DTLS_STATE_CONNECTED\n");
1804

1807  record_header, data, data_length) > 0) {
1808

1827  * followed by a ChangeCipherSpec and an encrypted Finished.
1828  */
1829  }
1830

1833  case DTLS_STATE_INIT: /* these states should not occur here */
1834  case DTLS_STATE_KEYEXCHANGE:
1835  default:

1838  }
