72 dtls_peer_t *
73 dtls_new_peer(const session_t *session) {
74  dtls_peer_t *peer;
75 
76  peer = dtls_malloc_peer();
77  if (peer) {
78  memset(peer, 0, sizeof(dtls_peer_t));
79  memcpy(&peer->session, session, sizeof(session_t));
80 
81 #ifndef NDEBUG
82  if (dtls_get_log_level() >= LOG_DEBUG) {
83  unsigned char addrbuf[72];
84  dsrv_print_addr(session, addrbuf, sizeof(addrbuf));
85  printf("dtls_new_peer: %s\n", addrbuf);
86  }
87 #endif
88  /* initially allow the NULL cipher */
89  CURRENT_CONFIG(peer)->cipher = TLS_NULL_WITH_NULL_NULL;
90 
91  /* initialize the handshake hash wrt. the hard-coded DTLS version */
92  debug("DTLSv12: initialize HASH_SHA256\n");
93  /* TLS 1.2: PRF(secret, label, seed) = P_<hash>(secret, label + seed) */
94  /* FIXME: we use the default SHA256 here, might need to support other
95  hash functions as well */
96  dtls_hash_init(&peer->hs_state.hs_hash);
97  }
98 
99  return peer;
100 }
