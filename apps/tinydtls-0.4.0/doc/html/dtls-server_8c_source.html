67 int
68 send_to_peer(struct dtls_context_t *ctx,
69  session_t *session, uint8 *data, size_t len) {
70 
71  struct uip_udp_conn *conn = (struct uip_udp_conn *)dtls_get_app_data(ctx);
72 
73  uip_ipaddr_copy(&conn->ripaddr, &session->addr);
74  conn->rport = session->port;
75 
76  uip_udp_packet_send(conn, data, len);
77 
78  /* Restore server connection to allow data from any node */
79  memset(&conn->ripaddr, 0, sizeof(server_conn->ripaddr));
80  memset(&conn->rport, 0, sizeof(conn->rport));
81 
82  return len;
83 }
