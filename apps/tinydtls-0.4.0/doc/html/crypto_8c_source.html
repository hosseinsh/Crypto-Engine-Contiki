82 size_t
83 dtls_p_hash(dtls_hashfunc_t h,
84  const unsigned char *key, size_t keylen,
85  const unsigned char *label, size_t labellen,
86  const unsigned char *random1, size_t random1len,
87  const unsigned char *random2, size_t random2len,
88  unsigned char *buf, size_t buflen) {
89  dtls_hmac_context_t *hmac_a, *hmac_p;
90 
91  unsigned char A[DTLS_HMAC_DIGEST_SIZE];
92  unsigned char tmp[DTLS_HMAC_DIGEST_SIZE];
93  size_t dlen; /* digest length */
94  size_t len = 0; /* result length */
95 
96  hmac_a = dtls_hmac_new(key, keylen);
97  if (!hmac_a)
98  return 0;
99 
100  /* calculate A(1) from A(0) == seed */
101  HMAC_UPDATE_SEED(hmac_a, label, labellen);
102  HMAC_UPDATE_SEED(hmac_a, random1, random1len);
103  HMAC_UPDATE_SEED(hmac_a, random2, random2len);
104 
105  dlen = dtls_hmac_finalize(hmac_a, A);
106 
107  hmac_p = dtls_hmac_new(key, keylen);
108  if (!hmac_p)
109  goto error;
110 
111  while (len + dlen < buflen) {
112 
113  /* FIXME: rewrite loop to avoid superflous call to dtls_hmac_init() */
114  dtls_hmac_init(hmac_p, key, keylen);
115  dtls_hmac_update(hmac_p, A, dlen);
116 
117  HMAC_UPDATE_SEED(hmac_p, label, labellen);
118  HMAC_UPDATE_SEED(hmac_p, random1, random1len);
119  HMAC_UPDATE_SEED(hmac_p, random2, random2len);
120 
121  len += dtls_hmac_finalize(hmac_p, tmp);
122  memcpy(buf, tmp, dlen);
123  buf += dlen;
124 
125  /* calculate A(i+1) */
126  dtls_hmac_init(hmac_a, key, keylen);
127  dtls_hmac_update(hmac_a, A, dlen);
128  dtls_hmac_finalize(hmac_a, A);
129  }
130 
131  dtls_hmac_init(hmac_p, key, keylen);
132  dtls_hmac_update(hmac_p, A, dlen);
133 
134  HMAC_UPDATE_SEED(hmac_p, label, labellen);
135  HMAC_UPDATE_SEED(hmac_p, random1, random1len);
136  HMAC_UPDATE_SEED(hmac_p, random2, random2len);
137 
138  dtls_hmac_finalize(hmac_p, tmp);
139  memcpy(buf, tmp, buflen - len);
140 
141  error:
142  dtls_hmac_free(hmac_a);
143  dtls_hmac_free(hmac_p);
144 
145  return buflen;
146 }

224 size_t
225 dtls_pre_master_secret(unsigned char *key, size_t keylen,
226  unsigned char *result) {
227  unsigned char *p = result;
228 
229  dtls_int_to_uint16(p, keylen);
230  p += sizeof(uint16);
231 
232  memset(p, 0, keylen);
233  p += keylen;
234 
235  memcpy(p, result, sizeof(uint16));
236  p += sizeof(uint16);
237 
238  memcpy(p, key, keylen);
239 
240  return (sizeof(uint16) + keylen) << 1;
241 }

250 dtls_cipher_context_t *
251 dtls_cipher_new(dtls_cipher_t cipher,
252  unsigned char *key, size_t keylen) {
253  dtls_cipher_context_t *cipher_context = NULL;
254 
255  cipher_context = dtls_cipher_context_new();
256  if (!cipher_context) {
257  warn("cannot allocate cipher_context\r\n");
258  return NULL;
259  }
260 
261  switch (cipher) {
262  case TLS_PSK_WITH_AES_128_CCM_8: {
263  aes128_ccm_t *ccm_ctx = &cipher_context->data;
264 
265  if (rijndael_set_key_enc_only(&ccm_ctx->ctx, key, 8 * keylen) < 0) {
266  /* cleanup everything in case the key has the wrong size */
267  warn("cannot set rijndael key\n");
268  goto error;
269  }
270  break;
271  }
272  default:
273  warn("unknown cipher %04x\n", cipher);
274  goto error;
275  }
276 
277  return cipher_context;
278  error:
279  dtls_cipher_context_free(cipher_context);
280  return NULL;
281 }
