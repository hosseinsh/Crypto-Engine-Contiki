59 try_send(struct dtls_context_t *ctx, session_t *dst) {

62  if (res >= 0) {

122 dtls_handle_read(dtls_context_t *ctx) {
123  static session_t session;
124 
125  if(uip_newdata()) {
126  uip_ipaddr_copy(&session.addr, &UIP_IP_BUF->srcipaddr);
127  session.port = UIP_UDP_BUF->srcport;
128  session.size = sizeof(session.addr) + sizeof(session.port);
129 
130  ((char *)uip_appdata)[uip_datalen()] = 0;
131  PRINTF("Client received message from ");
132  PRINT6ADDR(&session.addr);
133  PRINTF(":%d\n", uip_ntohs(session.port));
134 
135  dtls_handle_message(ctx, &session, uip_appdata, uip_datalen());
136  }

172 void
173 init_dtls(session_t *dst) {
174  static dtls_handler_t cb = {
175  .write = send_to_peer,
176  .read = read_from_peer,
177  .event = NULL,
178  .get_key = get_key
179  };
180  PRINTF("DTLS client started\n");
181 
182  print_local_addresses();
183 
184  dst->size = sizeof(dst->addr) + sizeof(dst->port);
185  dst->port = UIP_HTONS(20220);
186 
187  set_connection_address(&dst->addr);
188  client_conn = udp_new(&dst->addr, 0, NULL);
189  udp_bind(client_conn, dst->port);
190 
191  PRINTF("set connection address to ");
192  PRINT6ADDR(&dst->addr);
193  PRINTF(":%d\n", uip_ntohs(dst->port));
194 
195  set_log_level(LOG_DEBUG);
196 
197  dtls_context = dtls_new_context(client_conn);
198  if (dtls_context)
199  dtls_set_handler(dtls_context, &cb);
200 }

203 PROCESS_THREAD(udp_server_process, ev, data)
204 {
205  static int connected = 0;
206  static session_t dst;
207 
208  PROCESS_BEGIN();
209 
210  dtls_init();
211 
212  init_dtls(&dst);
213  serial_line_init();
214 
215  if (!dtls_context) {
216  dsrv_log(LOG_EMERG, "cannot create context\n");
217  PROCESS_EXIT();
218  }
219 
220  while(1) {
221  PROCESS_YIELD();
222  if(ev == tcpip_event) {
223  dtls_handle_read(dtls_context);
224  } else if (ev == serial_line_event_message) {
225  register size_t len = min(strlen(data), sizeof(buf) - buflen);
226  memcpy(buf + buflen, data, len);
227  buflen += len;
228  if (buflen < sizeof(buf) - 1)
229  buf[buflen++] = '\n'; /* serial event does not contain LF */
230  }
231 
232  if (buflen) {
233  if (!connected)
234  connected = dtls_connect(dtls_context, &dst) >= 0;
235 
236  try_send(dtls_context, &dst);
237  }
238  }
239 
240  PROCESS_END();
241 }
